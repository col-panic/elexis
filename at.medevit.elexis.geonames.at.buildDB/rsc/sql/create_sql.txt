CREATE TABLE GEMEINDEN (
	KENNZIFFER INTEGER,
	NAME VARCHAR(50) NOT NULL,
	GEMEINDECODE INTEGER NOT NULL CONSTRAINT gemeindecode_no_pk PRIMARY KEY,
	STATUS VARCHAR(10)
);

CREATE TABLE PLZ (
	PLZ INTEGER NOT NULL,
	KENNZIFFER INTEGER CONSTRAINT kennziffer_fk REFERENCES GEMEINDEN(GEMEINDECODE),
	HAUPTPLZ VARCHAR(1)
);

CREATE TABLE STRASSEN (
	GEMNR INTEGER CONSTRAINT gemnr_fk REFERENCES GEMEINDEN(GEMEINDECODE),
	PLZ INTEGER,
	STROFFI VARCHAR(50),
	SKZ INTEGER
);
CREATE INDEX Strassen_Plz ON STRASSEN(PLZ);
CREATE INDEX Strassen_PlzZuStrassen ON STRASSEN(PLZ, STROFFI);
